280  // Move key to left. Remove it if it fell off the edge
281  if((i.value().x -= K70_X_START) < 0)
282  i.remove();
283  }
284  // Remove the M buttons as well
285  map.remove("mr");
286  map.remove("m1");
287  map.remove("m2");
288  map.remove("m3");
289  // Done!
290  break;
291  }
292  case KeyMap::K65:{
293  // The K65 maps additionally remove the numpad and have a modified top row
294  map = getMap(KeyMap::K70, layout);
295  QMutableHashIterator<QString, Key> i(map);
296  while(i.hasNext()){
297  i.next();
298  if(i.value().x >= K65_WIDTH)
299  i.remove();
300  }
301  for(const Key* key = K65TopRow; key < K65TopRow + K65_TOP_COUNT; key++)
302  map[key->name] = *key;
303  // Done!
304  break;
305  }
306  case KeyMap::STRAFE:{
307  // The Strafe RGB maps are based on the K70 map minus the media keys
308  map = getMap(KeyMap::K70, layout);
309  //move light and lock right
310  map["light"].x=285 - K70_X_START;
311  //map["light"].hasLed=false;
312  map["lock"].x=297 - K70_X_START;
313  //map["lock"].hasLed=false;
314  // move everything right to make the space for the left sidelight
315  QMutableHashIterator<QString, Key> i(map);
316  while(i.hasNext()){
317  i.next();
318  i.value().x += KSTRAFE_X_START;
319  }
320  // Add Strafe lights and keys
321  map["lsidel"] = KStrafeKeys[0];
322  map["rsidel"] = KStrafeKeys[1];
323  map["logo"] = KStrafeKeys[2];
324  map["fn"] = KStrafeKeys[3];
325  map.remove("rwin");
326  // remove media controls
327  map.remove("mute");
328  map.remove("volup");
329  map.remove("voldn");
330  map.remove("stop");
331  map.remove("prev");
332  map.remove("play");
333  map.remove("next");
334  // Done!
335  break;
336  }
337  case KeyMap::M65:{
338  // M65 isn't a keyboard; all mouse maps are unique.
339  for(const Key* key = M65Keys; key < M65Keys + KEYCOUNT_M65; key++){
340  // Keyboard keys are written from the center because that's where the LEDs are, but the mouse buttons are odd shapes so they're
341  // written from the upper left
342  Key translatedKey = *key;
343  translatedKey.x += translatedKey.width / 2;
344  translatedKey.y += translatedKey.height / 2;
345  map[key->name] = translatedKey;
346  }
347  // Mice also have no layout patches - no other changes necessary
348  break;
349  }
350  case KeyMap::SABRE:{
351  // Scimitar mouse
352  for(const Key* key = SabreKeys; key < SabreKeys + KEYCOUNT_SABRE; key++){
353  // Like the M65, the keys are upper-left justified
354  Key translatedKey = *key;
355  translatedKey.x += translatedKey.width / 2;
356  translatedKey.y += translatedKey.height / 2;
357  map[key->name] = translatedKey;
358  }
359  break;
360  }
361  case KeyMap::SCIMITAR:{
362  // Scimitar mouse
363  for(const Key* key = ScimKeys; key < ScimKeys + KEYCOUNT_SCIM; key++){
364  Key translatedKey = *key;
365  translatedKey.x += translatedKey.width / 2;
366  translatedKey.y += translatedKey.height / 2;
367  map[key->name] = translatedKey;
368  }
369  break;
370  }
371  default:; // <- stop GCC from complaining
372  }
373  // Map is finished, return result
374  return map;
375 }
